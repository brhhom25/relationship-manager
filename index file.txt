function registerServiceWorker(){
  if(!('serviceWorker' in navigator))return;
  const swCode=`
const CACHE='rm-pro-v1';
const ASSETS=['/'];
self.addEventListener('install',e=>{e.waitUntil(caches.open(CACHE).then(c=>c.addAll(ASSETS)));self.skipWaiting();});
self.addEventListener('activate',e=>{e.waitUntil(caches.keys().then(ks=>Promise.all(ks.filter(k=>k!==CACHE).map(k=>caches.delete(k)))));self.clients.claim();});
self.addEventListener('fetch',e=>{if(e.request.method!=='GET')return;e.respondWith(caches.match(e.request).then(r=>r||fetch(e.request).then(resp=>{if(resp&&resp.status===200){const clone=resp.clone();caches.open(CACHE).then(c=>c.put(e.request,clone));}return resp;}).catch(()=>caches.match('/'))));});
self.addEventListener('sync',e=>{if(e.tag==='rm-sync'){e.waitUntil(self.clients.matchAll().then(clients=>clients.forEach(c=>c.postMessage({type:'sync'}))));}});
  `;
  const blob=new Blob([swCode],{type:'application/javascript'});
  const swUrl=URL.createObjectURL(blob);
  navigator.serviceWorker.register(swUrl).then(()=>{
    updateCacheStatus();
  }).catch(()=>{});
}
function updateCacheStatus(){
  const el=document.getElementById('cache-status');
  if(el)el.textContent='Service worker active';
}

let db=null;
function openIDB(){
  return new Promise((res,rej)=>{
    const req=indexedDB.open('RelationshipManagerPro',1);
    req.onupgradeneeded=e=>{
      const d=e.target.result;
      if(!d.objectStoreNames.contains('contacts'))d.createObjectStore('contacts',{keyPath:'id'});
      if(!d.objectStoreNames.contains('callRecords'))d.createObjectStore('callRecords',{keyPath:'id'});
      if(!d.objectStoreNames.contains('voiceJournals'))d.createObjectStore('voiceJournals',{keyPath:'id'});
      if(!d.objectStoreNames.contains('syncQueue'))d.createObjectStore('syncQueue',{keyPath:'id',autoIncrement:true});
    };
    req.onsuccess=e=>{db=e.target.result;res(db);};
    req.onerror=e=>rej(e);
  });
}
async function idbPut(store,item){
  if(!db)return;
  return new Promise((res,rej)=>{const t=db.transaction(store,'readwrite');t.objectStore(store).put(item);t.oncomplete=res;t.onerror=rej;});
}
async function idbGetAll(store){
  if(!db)return[];
  return new Promise((res,rej)=>{const t=db.transaction(store,'readonly');const req=t.objectStore(store).getAll();req.onsuccess=()=>res(req.result||[]);req.onerror=rej;});
}
async function idbDelete(store,key){
  if(!db)return;
  return new Promise((res,rej)=>{const t=db.transaction(store,'readwrite');t.objectStore(store).delete(key);t.oncomplete=res;t.onerror=rej;});
}

async function processSyncQueue(){
  if(!syncQueue.length||!isOnline)return;
  const n8nUrl=settings.n8n?.url;

    }
  }
  updateSyncQueueBadge();renderSyncQueueList();

}

function calcTraitsScore(c){if(!c.traits?.length)return null;return Math.round(c.traits.reduce((a,t)=>a+t.score,0)/c.traits.length);}
function calcIntScore(c){if(!c.interactions?.length)return null;return Math.round(c.interactions.reduce((a,i)=>a+(i.score||0),0)/c.interactions.length);}
function calcOverallScore(c){const ts=calcTraitsScore(c),is=calcIntScore(c);const parts=[];if(ts!==null)parts.push(ts);if(is!==null)parts.push(is);return parts.length?Math.round(parts.reduce((a,b)=>a+b,0)/parts.length):0;}
function calcHealthScore(c){
  let score=50;const daysSince=getDaysSinceContact(c);
  if(daysSince===0)score+=20;else if(daysSince<=7)score+=15;else if(daysSince<=14)score+=5;else if(daysSince<=30)score-=5;else if(daysSince<=60)score-=15;else score-=30;
  const os=calcOverallScore(c);score+=Math.round(os*0.3);
  const plans=(c.plans||[]).filter(p=>p.status==='pending').length;score+=Math.min(10,plans*3);
  return Math.max(0,Math.min(100,score));
}
function getDaysSinceContact(c){if(!c.interactions?.length)return 999;const last=c.interactions.reduce((a,b)=>new Date(b.date)>new Date(a.date)?b:a);return Math.floor((new Date()-new Date(last.date+'T12:00:00'))/86400000);}
function getLastIntDate(c){if(!c.interactions?.length)return'Never';const l=c.interactions.reduce((a,b)=>new Date(b.date)>new Date(a.date)?b:a);return fmtDate(l.date);}
function getTrend(c){const ints=c.interactions||[];if(ints.length<2)return'stable';const sorted=[...ints].sort((a,b)=>new Date(a.date)-new Date(b.date));const half=Math.ceil(sorted.length/2);const a1=arr=>arr.length?arr.reduce((a,b)=>a+b,0)/arr.length:0;const first=sorted.slice(0,half).map(i=>i.score||0);const sec=sorted.slice(half).map(i=>i.score||0);const diff=a1(sec)-a1(first);if(diff>10)return'improving';if(diff<-10)return'declining';return'stable';}

function initVoiceCommands(){
  const SpeechRecognition=window.SpeechRecognition||window.webkitSpeechRecognition;
  if(!SpeechRecognition){document.getElementById('voice-cmd-btn').style.display='none';return;}
  speechRecognition=new SpeechRecognition();
  speechRecognition.continuous=false;speechRecognition.interimResults=false;speechRecognition.lang='en-US';
  speechRecognition.onresult=e=>{
    const cmd=e.results[0][0].transcript.toLowerCase().trim();
    handleVoiceCommand(cmd);
  };
  speechRecognition.onerror=()=>stopListening();
  speechRecognition.onend=()=>stopListening();
  const btn=document.getElementById('voice-cmd-btn');
  btn.addEventListener('mousedown',startListening);
  btn.addEventListener('touchstart',e=>{e.preventDefault();startListening();});
  btn.addEventListener('mouseup',stopListening);
  btn.addEventListener('touchend',stopListening);
  // Double-click for help
  btn.addEventListener('dblclick',()=>{showVoiceHelp();});
}
function startListening(){
  if(!speechRecognition||voiceListening)return;

  document.getElementById('voice-cmd-btn').classList.add('listening');
  document.getElementById('voice-cmd-indicator').classList.add('show');
  document.getElementById('voice-cmd-text').textContent='Listening…';

}
function stopListening(){
  if(!speechRecognition||!voiceListening)return;
  voiceListening=false;
  document.getElementById('voice-cmd-btn').classList.remove('listening');
  setTimeout(()=>document.getElementById('voice-cmd-indicator').classList.remove('show'),1500);
  try{speechRecognition.stop();}catch{}
}

async function sendWhatsAppTemplate(toPhone,templateName,lang='en'){
  const phoneId=settings.whatsapp?.phoneId;const token=settings.whatsapp?.token;
  if(!phoneId||!token)throw new Error('WhatsApp not configured. Set Phone ID and Token in Settings → Integrations.');
  const url=`https://graph.facebook.com/v18.0/${phoneId}/messages`;
  const body={messaging_product:'whatsapp',to:toPhone,type:'template',template:{name:templateName,language:{code:lang}}};
  const r=await fetch(url,{method:'POST',headers:{Authorization:`Bearer ${token}`,'Content-Type':'application/json'},body:JSON.stringify(body)});
  if(!r.ok){const e=await r.json();throw new Error(`WhatsApp API: ${e.error?.message||r.status}`);}

function parseCSVFile(file){
  const reader=new FileReader();
  reader.onload=e=>{
    const text=e.target.result;
    const lines=text.split('\n').filter(l=>l.trim());
    if(!lines.length){showToast('Empty CSV file','error');return;}
    csvHeaders=lines[0].split(',').map(h=>h.trim().replace(/"/g,''));
    csvParsedData=lines.slice(1).map(line=>{
      const vals=line.match(/(".*?"|[^",]+)(?=\s*,|\s*$)/g)||line.split(',');
      return csvHeaders.reduce((obj,h,i)=>{obj[h]=(vals[i]||'').trim().replace(/"/g,'');return obj;},{});
    }).filter(r=>Object.values(r).some(v=>v));
    showCSVMapper();
  };
  reader.readAsText(file);
}

function parseVCFFile(file){
  const reader=new FileReader();
  reader.onload=e=>{
    const text=e.target.result;
    const vcards=text.split('BEGIN:VCARD').filter(v=>v.trim());
    let imported=0;
    vcards.forEach(vc=>{
      const getName=re=>vc.match(re)?.[1]?.trim()||'';
      const fnMatch=vc.match(/FN[^:]*:([^\r\n]+)/);
      const nMatch=vc.match(/N[;:][^\r\n]*/);
      let name=fnMatch?fnMatch[1].trim():'';
      if(!name&&nMatch){const parts=nMatch[0].split(':')[1]?.split(';')||[];name=[parts[1],parts[0]].filter(Boolean).join(' ');}
      if(!name)return;
      const telMatch=vc.match(/TEL[^:]*:([^\r\n]+)/);
      const emailMatch=vc.match(/EMAIL[^:]*:([^\r\n]+)/);
      const noteMatch=vc.match(/NOTE[^:]*:([^\r\n]+)/);
      const phone=(telMatch?.[1]||'').trim();
      const email=(emailMatch?.[1]||'').trim();
      if(contacts.find(c=>c.name===name||(phone&&c.phone===phone)))return;
      contacts.push({id:Date.now()+Math.random()+imported,name,phone,email,category:'personal',bestTime:'',checkinDays:'',notes:(noteMatch?.[1]||'').trim(),traits:[],interactions:[],plans:[],subjects:[],importantDays:[]});
      imported++;
    });
    saveData();renderContacts();
    document.getElementById('import-result').innerHTML=`<div style="background:#d1fae5;border:1px solid #6ee7b7;border-radius:8px;padding:12px;margin-top:10px;"><i class="fas fa-check-circle" style="color:#065f46;"></i> Imported <strong>${imported}</strong> contacts from vCard.</div>`;
    showToast(`Imported ${imported} from vCard!`,'success');
  };
  reader.readAsText(file);
}

function renderMessageThread(cid,channel){
  const c=contacts.find(x=>x.id===cid);if(!c)return;
  // Synthetic messages from interactions + demo
  if(!contactMessages[cid])contactMessages[cid]=[];
  const msgs=contactMessages[cid];
  const bubblesEl=document.getElementById('msg-bubble-wrap');
  if(!bubblesEl)return;
  document.getElementById('message-thread-ui').style.display='block';
  document.getElementById('msg-thread-title').textContent=c.name;
  document.getElementById('msg-thread-channel').textContent=channel||'app';
  document.getElementById('msg-thread-channel').className=`channel-badge ch-${channel||'telegram'}`;
  renderBubbles(msgs,bubblesEl);
  const sendBtn=document.getElementById('msg-send-btn');
  const input=document.getElementById('msg-compose-input');
  sendBtn.onclick=()=>sendMsgFromThread(cid,channel,input,bubblesEl);
  input.onkeydown=e=>{if(e.key==='Enter')sendMsgFromThread(cid,channel,input,bubblesEl);};

function renderBubbles(msgs,el){
  if(!msgs.length){el.innerHTML='<div style="text-align:center;padding:20px;color:#aaa;font-size:.85rem;">No messages yet — send one below!</div>';return;}
  el.innerHTML=msgs.map(m=>`<div class="msg-bubble ${m.dir}"><div class="msg-text ${m.dir}">${esc(m.text)}</div><div class="msg-meta">${fmtDate(m.date)} ${m.dir==='sent'?'✓✓':''}</div></div>`).join('');
  el.scrollTop=el.scrollHeight;
}
async function sendMsgFromThread(cid,channel,input,bubblesEl){
  const text=input.value.trim();if(!text)return;
  const c=contacts.find(x=>x.id===cid);if(!c)return;
  if(!contactMessages[cid])contactMessages[cid]=[];
  const msg={id:Date.now(),dir:'sent',text,date:today(),channel};
  contactMessages[cid].push(msg);
  input.value='';renderBubbles(contactMessages[cid],bubblesEl);
  // Try real send
  if(channel==='whatsapp'&&c.phone&&settings.whatsapp?.token){
    try{await sendWhatsAppText(c.phone,text);showToast('Sent via WhatsApp!','success');}
    catch(e){showToast('WA: '+e.message,'error');}
  } else if(channel==='telegram'&&settings.telegram?.botToken){
    const chatId=c.telegramChatId;
    if(chatId){try{await sendTelegramMessage(chatId,text);showToast('Sent via Telegram!','success');}catch(e){showToast('TG: '+e.message,'error');}}
    else{showToast('No Telegram chat ID for this contact. They need to start a chat with your bot first.','error');}
  } else{showToast('Message saved locally (configure API keys to send for real)','info');}
  addToSyncQueue('message_sent',{contactId:cid,text,channel});
}

  const cx=canvas.width/2,cy=canvas.height/2;
  ctx.beginPath();ctx.arc(cx,cy,22,0,Math.PI*2);ctx.fillStyle='#7C3AED';ctx.fill();
  ctx.fillStyle='white';ctx.font='bold 11px DM Sans';ctx.textAlign='center';ctx.textBaseline='middle';ctx.fillText('YOU',cx,cy);
  const maxR=Math.min(cx,cy)-40;
  contacts.slice(0,20).forEach((c,i)=>{

async function runAIBreakdown(recId){
  const rec=callRecords.find(r=>r.id===recId);if(!rec)return;
  const area=document.getElementById(`breakdown-area-${recId}`);if(!area)return;
  if(!gKey()){area.innerHTML='<div style="padding:12px;background:#fee2e2;border-radius:8px;color:#991b1b;"><i class="fas fa-exclamation-triangle"></i> No Groq API key. Add it in Settings → AI.</div>';return;}
  area.innerHTML='<div class="ai-loading"><div class="ai-spinner"></div><div style="color:var(--ai);font-size:.85rem;">Step 1: Transcribing…</div></div>';
  let transcript=rec.transcript||'';
  try{
    if(!transcript){
      if(!rec.audioDataUrl){area.innerHTML='<div style="padding:12px;background:#fef3c7;border-radius:8px;color:#92400e;">Audio not available. Re-upload to transcribe.</div>';return;}
      const res=await fetch(rec.audioDataUrl);const blob=await res.blob();
      transcript=await groqTranscribe(new File([blob],rec.filename||'audio.mp3',{type:blob.type}));
      rec.transcript=transcript;saveData();
    }
    area.innerHTML='<div class="ai-loading"><div class="ai-spinner"></div><div style="color:var(--ai);font-size:.85rem;">Step 2: Analyzing…</div></div>';
    const bd=await groqBreakdown(transcript);
    if(rec.markerActionItems?.length)bd.action_items=(bd.action_items||[]).concat(rec.markerActionItems);
    rec.breakdown=bd;saveData();
    area.innerHTML=renderBreakdownHtml(recId,bd);
    triggerN8n('breakdown',{recId,title:rec.title,sentiment:bd.sentiment});
    showToast('AI breakdown complete!','success');
  }catch(err){area.innerHTML=`<div style="padding:12px;background:#fee2e2;border-radius:8px;color:#991b1b;"><i class="fas fa-exclamation-triangle"></i> ${esc(err.message)}</div>`;}
}

function expandRecurring(events){
  const out=[];const lim=new Date();lim.setFullYear(lim.getFullYear()+2);
  events.forEach(ev=>{out.push(ev);if(!ev.recurring)return;const b=new Date(ev.date+'T12:00:00');let n=new Date(b);for(let i=0;i<2;i++){n=new Date(n);n.setFullYear(n.getFullYear()+1);if(n>lim)break;out.push({...ev,date:n.toISOString().split('T')[0],_rec:true});}});
  return out;
}
function getAllCalEvents(ft,fm){
  let evs=[];
  contacts.forEach(c=>{
    (c.plans||[]).forEach(p=>evs.push({...p,type:'plan',contact:c.name,contactId:c.id}));
    (c.interactions||[]).forEach(i=>evs.push({...i,contact:c.name,contactId:c.id}));
    expandRecurring((c.importantDays||[]).map(d=>({...d,type:'important',contact:c.name,contactId:c.id}))).forEach(d=>evs.push(d));
  });
  if(ft&&ft!=='all'){if(ft==='interactions')evs=evs.filter(e=>e.type!=='plan'&&e.type!=='important');else if(ft==='plans')evs=evs.filter(e=>e.type==='plan');else if(ft==='important')evs=evs.filter(e=>e.type==='important');}
  if(fm)evs=evs.filter(e=>e.date?.startsWith(fm));
  return evs.sort((a,b)=>new Date(a.date)-new Date(b.date));
}

function setupInlineEditing(){
  document.querySelectorAll('.editable').forEach(el=>{
    if(el._setup)return;el._setup=true;
    el.addEventListener('click',function(){
      if(this.classList.contains('editing'))return;
      const field=this.getAttribute('data-field');const cv=this.textContent;this.classList.add('editing');
      let inp;
      if(field==='notes'){inp=document.createElement('textarea');inp.style.minHeight='70px';}
      else if(field==='category'){inp=document.createElement('select');['family','personal','professional','friend','business','other'].forEach(v=>{const o=document.createElement('option');o.value=v;o.textContent=v;if(v===cv)o.selected=true;inp.appendChild(o);});}
      else{inp=document.createElement('input');inp.type='text';}
      inp.className='editable-input';inp.value=cv;this.innerHTML='';this.appendChild(inp);inp.focus();
      const save=()=>{const nv=inp.value;this.classList.remove('editing');this.textContent=nv;if(currentContactId){const c=contacts.find(x=>x.id===currentContactId);if(c){c[field]=nv;saveData();renderContacts();}}};
      inp.addEventListener('blur',save);
      inp.addEventListener('keydown',e=>{if(e.key==='Enter'&&field!=='notes')save();if(e.key==='Escape'){this.classList.remove('editing');this.textContent=cv;}});
    });
  });
}
